\begin{frame}[t]{Estimator: From \cite{DLP12} to Our Generalization}
\small
\begin{columns}[T]
\begin{column}{0.47\textwidth}
\textbf{\cite{DLP12} (special case)}

\medskip
Pricing residual at inception $t_{0,i}$:
\[
\epsilon_i^{\text{DLP}}(\theta)=\sum_{t=0}^T \Psi_{t}(\theta)\,CF_{i,t}
\]
Stochastic Discount Factor (SDF):
{\tiny
\[
\Psi_{t}(\alpha, \beta) =
\frac{1}{
	\prod_{s=t_{0,i}+1}^{t}
	\left( 
	1 + R_{\textbf{riskfree}, s} + \alpha + \beta R_{\textbf{MKT}, s}
	\right)
} 
\]
}
Cross-sectional estimator:
\[
\hat\theta_{\text{DLP}}=\arg\min_{\theta\in\Theta}\frac{1}{n}\sum_{i=1}^n\!\left(\epsilon_i^{\text{DLP}}\right)^2
\]

\end{column}
\hfill
\vrule{}
\hfill
\begin{column}{0.47\textwidth}
\textbf{Our generalization}

\medskip
Pricing residual at \emph{any} date $\tau$:
\[
\epsilon_{\tau,i}(\theta)=\sum_{t=0}^T \frac{\Psi_{t}(\theta)}{\textcolor{cyan}{\Psi_{\tau}(\theta)} }\,CF_{i,t}
\]

Average over \textcolor{cyan}{compounding} \textcolor{blue}{horizons $\mathcal{T}_i$}:
\[
\bar\epsilon_i(\theta)=\frac{1}{\textcolor{blue}{ |\mathcal{T}_i|}}\sum_{\tau \in \textcolor{blue}{\mathcal{T}_i}}\epsilon_{\tau,i}(\theta)
\]

Nonlinear least-mean-distance:
\[
\hat\theta=\arg\min_{\theta\in\Theta}\frac{1}{n}\sum_{i=1}^n L\!\left(\bar\epsilon_i(\theta)\right)
\]


\end{column}
\end{columns}

\vspace{0.3em}
{\tiny \textbf{Variables:} $i$: fund index, $t$: cash-flow time, $\tau$: compounding date, $CF_{i,t}$: net cash flow, $\Psi_t(\theta)$: SDF, $\frac{\Psi_t}{\Psi_\tau}$: SDF ratio, $\theta\!\in\!\Theta$: parameters, $\mathcal{T}_i$: compounding-date set, $L$: loss function, $n$: number of units. \textbf{Nesting:}\ set $\mathcal{T}_i=\{t_{0,i}\}$ and $L(x)=x^2$ $\;\Longrightarrow\;$ recovers \cite{DLP12}.}

\end{frame}

\begin{frame}[t]{Estimator: Net Present Value (NPV) vs Net Future Value (NFV)}
	\begin{itemize}
		\item NPV-only discounting (fund inception) is theoretically unbiased.
		\item NFV: Adding \textcolor{cyan}{compounding} \textcolor{blue}{horizons $>0$} introduces a \textbf{bias term}. \\
		$\to$ Definition ``Horizon'': We average over all dates up to \textcolor{blue}{$\max (\mathcal{T})$}.
		\item \textbf{Spoiler:} Finite-sample performance can improve when averaging across future dates.
	\end{itemize}

\end{frame}

  	\begin{itemize}

  \item \textbf{Compare sample geometry:} more vintages ($V$) vs larger within-vintage size ($n/V$).
  \item \textbf{Compare SDF model forms:} Single-factor vs multi-factor models.
\end{itemize}

\begin{frame}{Simulation: Synthesis}
\begin{enumerate}
  \item Use portfolio formation to stabilize estimation.
  \item Increase number of funds per vintage (if possible).
  \item Use simple SDF models given limited amount of PE data.
  \item \textcolor{cyan}{Compounding} \textcolor{blue}{horizon choice (size of $\mathcal{T}$)} is a control for finite-sample bias. \\
  $\to$ \textcolor{blue}{$\mathcal{T}$-averaging horizons} from 10 to 20 years yield best results in our examples. \\
  $\to$ Not overly precise control!
\end{enumerate}

\vspace{0.4em}
\textbf{Interpretation:} practical estimator quality is dominated by finite-sample bias-variance tradeoffs.
\end{frame}

\begin{frame}{Motivation: Why Measuring Risk Is Hard in Private Markets}
	\begin{itemize}
		\item PE funds generate \textbf{cash flow sequences}, not continuously traded returns.
		\item Fund lives overlap across vintages, creating dependence beyond standard panel assumptions.
		\item Fund valuation relies on reported NAVs, which can be stale/smoothed.
		\item Standard return-based factor models are not directly applicable.
	\end{itemize}
	
	\vspace{0.4em}
	\textbf{Implication:} we need a cash-flow-native SDF estimator with robust dependence-aware inference.
\end{frame}

\begin{frame}{Estimator: Starting Point \cite{DLP12}}
	\small
	For each fund $i$, \cite{DLP12} imposes a zero-NPV pricing condition at inception date $t_{0,i}$:
	\[
	\epsilon_i^{\text{DLP}}(\theta)=\sum_{t=0}^T \Psi_{t}(\theta)\,CF_{i,t}
	\]
	
	Cross-sectional estimator:
	\[
	\hat\theta_{\text{DLP}}=\arg\min_{\theta\in\Theta}\frac{1}{n}\sum_{i=1}^n\left(\epsilon_i^{\text{DLP}}(\theta)\right)^2
	\]
	
	\vspace{1.5em}
	\textbf{We propose generalization} that nests the \cite{DLP12} estimator as special case. \\
	\vspace{3.5em}
	
	{\small Variables: $i$ fund/portfolio index, $t$ cash-flow time, $CF_{i,t}$ net cash flow, $\theta$ parameter vector, $\Theta$ parameter space, $n$ number of units, sample end date $T$.} \\
	\textbf{Important:} $\Psi_{t}(\theta)$ Stochastic Discount Factor (SDF) from $t$ to time $t_{0,i}$ (=fund inception date).
\end{frame}

\begin{frame}{Estimator: Our generalization}
	For fund $i$ and discounting date $\tau$:
	\[
	\epsilon_{\tau,i}(\theta)=\sum_{t=0}^T \frac{\Psi_{t}(\theta)}{\Psi_{\tau}(\theta)} CF_{i,t}
	\]
	
	Average over selected compounding dates $\mathcal{T}_i$ (\textcolor{blue}{compounding horizons}):
	\[
	\bar\epsilon_i(\theta)=\frac{1}{|\mathcal{T}_i|}\sum_{\tau\in\mathcal{T}_i}\epsilon_{\tau,i}(\theta)
	\]
	
	Estimate parameters by nonlinear least-mean-distance:
	\[
	\hat\theta=\arg\min_{\theta\in\Theta}\left(\frac{1}{n}\sum_{i=1}^n L(\bar\epsilon_i(\theta))\right),\quad L(x)=x^2
	\]
	\vspace{0.25em}
	\tiny Variables: $i$ fund/portfolio index, $t$ cash-flow time, $\tau$ compounding date, $CF_{i,t}$ net cash flow, $\frac{\Psi_{t}(\theta)}{\Psi_{\tau}(\theta)}$ SDF ratio, $\theta$ parameter vector, $\mathcal{T}_i$ compounding-date set, $L$ loss function.
\end{frame}
